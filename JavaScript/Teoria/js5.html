Składnia: 


var Auta = [
    {marka: "Volvo", rocznik: 2016},
    {marka: "Audi", rocznik: 2001},
    {marka: "BMW", rocznik: 2010}  
];

wyśAuta();

function SortowanieMarka() {
   Auta.sort(function(a, b){
      var x = a.marka.toLowerCase();
      var y = b.marka.toLowerCase();
      if (x < y) {return -1;}
      if (x > y) {return 1;}
      return 0;
   });
   wyśAuta();
}

function wyśAuta() {
   document.getElementById("sortMarka").innerHTML =

" +
