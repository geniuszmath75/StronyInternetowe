2. finally:

a) Składnia ogólna:

try {
   // Blok kodu do sprawdzenia
}
catch(err) {
   // Blok kodu do wychwycenia błedów
}
finally {
   // Blok kodu do wykonania niezależnie od wyniku try/catch
}

b) Przykład:

Składnia: 

function Sprawdzenie() {

   var message, x;
   message = document.getElementById("p01");
   message.innerHTML = "";
   x = document.getElementById("f").value;

   try {
   	   if (x == "") throw "nie została podana";
   	   if (isNaN(x)) throw "nie jest liczbą";
   	   x = Number(x);
       if (x > 10) throw "jest za duża";
   	   if (x < 5) throw "jest za mała";
   }
   catch (err) {
   	   message.innerHTML = "Liczba " + err;
   }
   finally {
   	   document.getElementById("f").value = "";
   	}
}
